		random_realm_province = {

	option = {
		name = EVTOPTA_ARTIFACTS_2 # Let us go to the village!

					narrative_event = { id = artifacts.5 days = 30 random = 30 } # Sword exists

		}
	}

		name = EVTOPTB_ARTIFACTS_2 # I have no time for this!
		
		prestige = -10
		
		ai_chance = {
			factor = 0
		}
	}
}

### Artifact transfer on succession for special artefacts

# The Heirloom Seal of the Realm sticks with China/a pretender empire
character_event = {
	id = artifacts.159
	hide_window = yes
	is_triggered_only = yes
	
	only_playable = yes
	
	trigger = {
		FROMFROM = {
			has_artifact = heirloom_seal_of_the_realm
		}
		FROM = {
			OR = {
				title = e_china
				has_title_flag = pretender_chinese_empire
			}
		}
	}
	
	immediate = {
		FROMFROM = {
			any_artifact = {
				limit = {
					has_artifact_flag = heirloom_seal_of_the_realm
				}
				transfer_artifact = {
					from = PREV
					to = ROOT
				}
			}
		}
	}
}
